when http server listen method:"post" path:"/upload" as req
    video = req.files["upload"]
    topics = machinebox/videobox detect input:video
    res = psql insert table:"videos" data:{..stuff from above..}
    video = someOpenSourceCompressor compress video:video
    storage put content:video dest:"/{res['uuid']}.mp4"
    req write content:"/{res['uuid']}.mp4"
